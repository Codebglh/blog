# 导入 pymysql 模块

import pymysql

# 连接 database

conn = pymysql.connect(host=“localhost”, user=“root”,password=“liuhonga”,database=“img”,charset=“utf8”)

# 得到一个可以执行 SQL 语句的光标对象

cursor = conn.cursor()

# 查询数据的 SQL 语句

sql = “SELECT url,_image from img;”

# 执行 SQL 语句

cursor.execute(sql)

# 获取多条查询数据

ret = cursor.fetchall()
cursor.close()
conn.close()
print(ret)
